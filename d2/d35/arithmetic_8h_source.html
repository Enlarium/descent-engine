296DESCENT_BUILTIN_API int
297uadd_carry_64(uint64_t a, uint64_t b, int carry_in, uint64_t *result) {
298#if DESCENT_HAS_BUILTIN(__builtin_addcll)
299 unsigned long long carry_out = 0;
300 *result = __builtin_addcll((unsigned long long) a, (unsigned long long) b, !!(unsigned long long) carry_in, &carry_out);
301 return (uint64_t) carry_out;
302#elif defined(DESCENT_PLATFORM_COMPILER_MSVC) && (_MSC_VER >= 1800)
303 return _addcarry_u64((unsigned char) !!carry_in, a, b, result);
304#else
305 uint64_t temp = a + b + !!carry_in;
306 *result = temp;
307 return (temp < a) || (carry_in && temp == a + b);
308#endif
309}

628// Partial 128-Bit Multiplication

637DESCENT_BUILTIN_API void
638umul_partial_128(const uint64_t a, const uint64_t b, uint64_t *high, uint64_t *low) {
639#if defined(DESCENT_PLATFORM_COMPILER_MSVC) && (DESCENT_PLATFORM_ARCHITECTURE_SIZE == 64) && (_MSC_VER >= 1400)
640 *low = _umul128(a, b, high);
641#elif __SIZEOF_INT128__
642 __uint128_t result = (__uint128_t) a * (__uint128_t) b;
643 *high = (uint64_t) (result >> 64);
644 *low = (uint64_t) result;
645#else
646 DESCENT_BUILTIN_CONST uint64_t MASK = 0xFFFFFFFFull;
647
648 uint64_t a_low = a & MASK;
649 uint64_t b_low = b & MASK;
650 uint64_t a_high = (a >> 32) & MASK;
651 uint64_t b_high = (b >> 32) & MASK;
652
653 uint64_t product_0 = a_low * b_low;
654 uint64_t product_1 = a_low * b_high;
655 uint64_t product_2 = a_high * b_low;
656 uint64_t product_3 = a_high * b_high;
657
658 uint64_t temp = (product_0 >> 32) + (product_1 & MASK) + (product_2 & MASK);
659 uint64_t carry = temp >> 32;
660
661 *low = (product_0 & MASK) | (temp << 32);
662 *high = product_3 + (product_2 >> 32) + (product_1 >> 32) + carry;
663#endif
664}

673DESCENT_BUILTIN_API void
674mul_partial_128(const int64_t a, const int64_t b, int64_t *high, int64_t *low) {
675#if defined(DESCENT_PLATFORM_COMPILER_MSVC) && (DESCENT_PLATFORM_ARCHITECTURE_SIZE == 64) && (_MSC_VER >= 1400)
676 *low = _mul128(a, b, high);
677#elif __SIZEOF_INT128__
678 __int128_t result = (__int128_t) a * (__int128_t) b;
679 *high = (int64_t) (result >> 64);
680 *low = (int64_t) result;
681#else
682 int negative = (a < 0) ^ (b < 0);
683
684 uint64_t abs_a = a < 0 ? -(uint64_t)a : (uint64_t)a;
685 uint64_t abs_b = b < 0 ? -(uint64_t)b : (uint64_t)b;
686
687 uint64_t abs_high = 0;
688 uint64_t abs_low = 0;
689
690 umul_partial_128(abs_a, abs_b, &abs_high, &abs_low);
691
692 if (negative) {
693 abs_low = ~abs_low + 1;
694 abs_high = ~abs_high + (abs_low == 0 ? 1 : 0);
695 }
696
697 *low = (int64_t) abs_low;
698 *high = (int64_t) abs_high;
699#endif
700}

740// Round to the next highest power of two
741DESCENT_BUILTIN_API uint64_t
742pow2ceil_64(uint64_t v) {
743 // Increment twice if v == 0 so that the next power is 1, not 2. Overflow will return 0.
744 uint64_t increment = (v == 0) + 1;
745 v--;
746 v |= v >> 1;
747 v |= v >> 2;
748 v |= v >> 4;
749 v |= v >> 8;
750 v |= v >> 16;
751 v |= v >> 32;
752 return v + increment;
753}
