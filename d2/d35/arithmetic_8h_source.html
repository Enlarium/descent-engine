76add_exflow_64(int64_t a, int64_t b, int64_t *result) {
77#if DESCENT_HAS_BUILTIN(__builtin_saddll_overflow)

79#else
80 *result = (int64_t) ((uint64_t) a + (uint64_t) b);

82#endif

295DESCENT_BUILTIN_API int
296uadd_carry_64(uint64_t a, uint64_t b, int carry_in, uint64_t *result) {
297#if DESCENT_HAS_BUILTIN(__builtin_addcll)
298 unsigned long long carry_out = 0;
299 *result = __builtin_addcll((unsigned long long) a, (unsigned long long) b, !!(unsigned long long) carry_in, &carry_out);
300 return (uint64_t) carry_out;
301#elif defined(DESCENT_PLATFORM_COMPILER_MSVC) && (_MSC_VER >= 1800)
302 return _addcarry_u64((unsigned char) !!carry_in, a, b, result);
303#else
304 uint64_t temp = a + b + !!carry_in;
305 *result = temp;
306 return (temp < a) || (carry_in && temp == a + b);
307#endif
308}

627// Partial 128-Bit Multiplication
628
636DESCENT_BUILTIN_API void
637umul_partial_128(const uint64_t a, const uint64_t b, uint64_t *high, uint64_t *low) {
638#if defined(DESCENT_PLATFORM_COMPILER_MSVC) && (DESCENT_PLATFORM_ARCHITECTURE_SIZE == 64) && (_MSC_VER >= 1400)
639 *low = _umul128(a, b, high);
640#elif __SIZEOF_INT128__
641 __uint128_t result = (__uint128_t) a * (__uint128_t) b;
642 *high = (uint64_t) (result >> 64);
643 *low = (uint64_t) result;
644#else
645 DESCENT_BUILTIN_CONST uint64_t MASK = 0xFFFFFFFFull;
646
647 uint64_t a_low = a & MASK;
648 uint64_t b_low = b & MASK;
649 uint64_t a_high = (a >> 32) & MASK;
650 uint64_t b_high = (b >> 32) & MASK;
651
652 uint64_t product_0 = a_low * b_low;
653 uint64_t product_1 = a_low * b_high;
654 uint64_t product_2 = a_high * b_low;
655 uint64_t product_3 = a_high * b_high;
656
657 uint64_t temp = (product_0 >> 32) + (product_1 & MASK) + (product_2 & MASK);
658 uint64_t carry = temp >> 32;
659
660 *low = (product_0 & MASK) | (temp << 32);
661 *high = product_3 + (product_2 >> 32) + (product_1 >> 32) + carry;
662#endif
663}

672DESCENT_BUILTIN_API void
673mul_partial_128(const int64_t a, const int64_t b, int64_t *high, int64_t *low) {
674#if defined(DESCENT_PLATFORM_COMPILER_MSVC) && (DESCENT_PLATFORM_ARCHITECTURE_SIZE == 64) && (_MSC_VER >= 1400)
675 *low = _mul128(a, b, high);
676#elif __SIZEOF_INT128__
677 __int128_t result = (__int128_t) a * (__int128_t) b;
678 *high = (int64_t) (result >> 64);
679 *low = (int64_t) result;
680#else
681 int negative = (a < 0) ^ (b < 0);
682
683 uint64_t abs_a = a < 0 ? -(uint64_t)a : (uint64_t)a;
684 uint64_t abs_b = b < 0 ? -(uint64_t)b : (uint64_t)b;
685
686 uint64_t abs_high = 0;
687 uint64_t abs_low = 0;
688
689 umul_partial_128(abs_a, abs_b, &abs_high, &abs_low);
690
691 if (negative) {
692 abs_low = ~abs_low + 1;
693 abs_high = ~abs_high + (abs_low == 0 ? 1 : 0);
694 }
695
696 *low = (int64_t) abs_low;
697 *high = (int64_t) abs_high;
698#endif
699}

739// Round to the next highest power of two
740DESCENT_BUILTIN_API uint64_t
741pow2ceil_64(uint64_t v) {
742 // Increment twice if v == 0 so that the next power is 1, not 2. Overflow will return 0.
743 uint64_t increment = (v == 0) + 1;
744 v--;
745 v |= v >> 1;
746 v |= v >> 2;
747 v |= v >> 4;
748 v |= v >> 8;
749 v |= v >> 16;
750 v |= v >> 32;
751 return v + increment;
752}
