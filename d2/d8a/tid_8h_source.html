51static inline thread_id tid_generate_unique(unsigned int index) {
52 // index is zero-based, maps to bit [TID_UNIQUE_INDEX_MIN + index]
53 if (index >= DESCENT_UNIQUE_THREAD_COUNT_MAX) return TID_NONE;
54 return 1ull << (TID_UNIQUE_INDEX_MIN + index);

82static inline bool tid_is_unique(thread_id t) {
83 // This checks that only one bit is set
84 // - "(t & TID_UNIQUE_MASK)" is true if "t" is nonzero and within the mask
85 // - "t & (t - 1)" is equal to "t" with the least significant set bit cleared
86 // - "!(t & (t - 1))" is therefore true if one or zero bits were set
87 return (t & TID_UNIQUE_MASK) && !(t & (t - 1));
88}

95static inline bool tid_is_worker(thread_id t) {
96 return (t & TID_WORKER_MASK) && !(t & (t - 1));
97}

105static inline bool tid_is_managed(thread_id t) {
106 return (t & (TID_MAIN | TID_UNIQUE_MASK | TID_WORKER_MASK)) && !(t & (t - 1));
107}
