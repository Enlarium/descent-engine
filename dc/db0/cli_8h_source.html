77static inline int cli_is_valid_short_name(char c) {
78 if (c == '!') return 1;
79 if (c >= '#' && c <= '&') return 1;
80 if (c >= '0' && c <= '9') return 1;
81 if (c >= '?' && c <= 'Z') return 1;
82 if (c >= 'a' && c <= 'z') return 1;
83
84 return 0;
85}

109static inline int cli_is_long_option(CLI_Parameter p) {
110 return (
111 p.action != NULL &&
112 p.name_long != NULL &&
113 (p.name_short == '\0' || cli_is_valid_short_name(p.name_short)) &&
114 p.parameters == NULL
115 );

124static inline int cli_is_short_option(CLI_Parameter p) {
125 return (
126 p.action != NULL &&
127 cli_is_valid_short_name(p.name_short) &&
128 p.parameters == NULL
129 );
130}

138static inline int cli_is_option(CLI_Parameter p) {
139 return cli_is_long_option(p) || cli_is_short_option(p);

148static inline int cli_is_positional(CLI_Parameter p) {

152 p.name_short == '\0' &&
153 p.position != 0 &&
154 p.position <= DESCENT_CLI_MAX_POSITIONALS &&

156 );
