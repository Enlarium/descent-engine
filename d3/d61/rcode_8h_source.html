26/*
27Codes less than or equal to 0 have special meanings reserved by the library.
28Positive values are user-defined and unregulated, but advised to follow this scheme.
29If user-defined values follow this scheme, they can use functions defined here for their own
30returns, otherwise they cannot.
31
32Severity:
33- 0: Operation fully succeeded; nothing special to report.
34- INFO: Operation succeeded, but carries a state the caller may need to inspect.
35 - *Only use for functions where the caller is expected to handle nuanced state.*
36 - Do not use INFO for trivial informational messages.
37- WARN: Operation partially succeeded, and the caller should check how it impacts them.
38 - *Only use for functions where the function has different levels of success that the
39 caller is expected to accept and handle*
40 - Do not use INFO for irrelevant state warnings or states that cannot be reasonably
41 continued from.
42- ERROR: Operation failed, and the caller needs to check what went wrong.
43- FATAL: Operation failed, and the caller needs to clean up and shut down.
44
45INFO and WARN cannot be generic or unknown, they MUST contain discerning information about the state
46resulting from the output.
47
48Fatal means "somebody messed the system up badly enough we cannot continue"
49
50Most codes are going to be errors
51*/

69#define RCODE_ORIGIN_WIDTH 1u
70#define RCODE_SEVERITY_WIDTH 2u
71#define RCODE_MODULE_WIDTH 13u
72#define RCODE_CODE_WIDTH 16u
73
74#define RCODE_ORIGIN_MASK ((1u << RCODE_ORIGIN_WIDTH) - 1)
75#define RCODE_SEVERITY_MASK ((1u << RCODE_SEVERITY_WIDTH) - 1)
76#define RCODE_MODULE_MASK ((1u << RCODE_MODULE_WIDTH) - 1)
77#define RCODE_CODE_MASK ((1u << RCODE_CODE_WIDTH) - 1)
78
79#define RCODE_ORIGIN_SHIFT (RCODE_WIDTH - RCODE_ORIGIN_WIDTH)
80#define RCODE_SEVERITY_SHIFT (RCODE_ORIGIN_SHIFT - RCODE_SEVERITY_WIDTH)
81#define RCODE_MODULE_SHIFT (RCODE_SEVERITY_SHIFT - RCODE_MODULE_WIDTH)
82#define RCODE_CODE_SHIFT (RCODE_MODULE_SHIFT - RCODE_CODE_WIDTH)
85#define RCODE_ORIGIN(rc) ((((uint32_t)rc) >> RCODE_ORIGIN_SHIFT) & RCODE_ORIGIN_MASK)
86#define RCODE_SEVERITY(rc) ((((uint32_t)rc) >> RCODE_SEVERITY_SHIFT) & RCODE_SEVERITY_MASK)
87#define RCODE_MODULE(rc) ((((uint32_t)rc) >> RCODE_MODULE_SHIFT) & RCODE_MODULE_MASK)
88#define RCODE_CODE(rc) ((((uint32_t)rc) >> RCODE_CODE_SHIFT) & RCODE_CODE_MASK)
89
90#define RCODE(origin, severity, module, code) ((int32_t)( \
91 ((origin & RCODE_ORIGIN_MASK) << RCODE_ORIGIN_SHIFT) | \
92 ((severity & RCODE_SEVERITY_MASK) << RCODE_SEVERITY_SHIFT) | \
93 ((module & RCODE_MODULE_MASK) << RCODE_MODULE_SHIFT) | \
94 ((code & RCODE_CODE_MASK) << RCODE_CODE_SHIFT) \
95))
96

181_Static_assert(RCODE_WIDTH == 32u, "rcode layout error: width must be 32");
182_Static_assert(sizeof(int) * CHAR_BIT >= RCODE_WIDTH, "rcode requires int to be at least 32 bits");
183_Static_assert(RCODE_CODE_SHIFT == 0u, "rcode layout error: code shift must be zero");
184_Static_assert(RC_USER_MIN == 1u, "rcode layout error: user min incorrect");
185_Static_assert(RCODE_ORIGIN(RC_USER_MIN) == RC_USER, "rcode layout error: user min incorrect");
186_Static_assert(RCODE_ORIGIN(RC_USER_MAX) == RC_USER, "rcode layout error: user max incorrect");
187_Static_assert(RCODE_ORIGIN(RC_USER_MAX + 1) == RC_DESCENT, "rcode layout error: user max is not max");
188_Static_assert(RCODE_ORIGIN(RC_DESCENT_MIN - 1) == RC_USER, "rcode layout error: descent min is not min");
189_Static_assert(RCODE_ORIGIN(RC_DESCENT_MIN) == RC_DESCENT, "rcode layout error: descent min incorrect");
190_Static_assert(RCODE_ORIGIN(RC_DESCENT_MAX) == RC_DESCENT, "rcode layout error: descent max incorrect");
191_Static_assert(RC_DESCENT_MAX == -0x00000001, "rcode layout error: descent max incorrect");
192_Static_assert(DESCENT_SUCCESS == 0, "DESCENT_SUCCESS must equal 0");
