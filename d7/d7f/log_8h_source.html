28typedef enum {
29 LOG_LEVEL_TRACE = 0x01,
30 LOG_LEVEL_INFO = 0x02,
31 LOG_LEVEL_DEBUG = 0x04,
32 LOG_LEVEL_WARN = 0x08,
33 LOG_LEVEL_ERROR = 0x10,
34 LOG_LEVEL_FATAL = 0x20,
35 LOG_LEVEL_DEFAULT = LOG_LEVEL_WARN | LOG_LEVEL_ERROR | LOG_LEVEL_FATAL,
36 LOG_LEVEL_ALL = LOG_LEVEL_TRACE | LOG_LEVEL_INFO | LOG_LEVEL_DEBUG | LOG_LEVEL_DEFAULT
37} LogLevel;

85static inline LogSinkHandle log_sink_handle(DescentModule m, int sink) {
86 return (LogSinkHandle) {.module = m, .sink = sink};
87}

124// Puts message onto queue. Limited to 256 characters, will truncate if too long.
125// Truncation is not failure, but will return a warning code.
126int log_message(DescentModule m, LogLevel l, const char *fmt, ...);

128// Puts message onto queue. Limited to 256 characters, will truncate if too long.
129// Truncation is not failure, but will return a warning code.
130int log_submit(DescentModule m, LogLevel l, const char *fmt, va_list args);

132// Writes a message from the queue
