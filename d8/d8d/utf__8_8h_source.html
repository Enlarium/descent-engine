24static inline size_t utf8_char_length(unsigned char c) {
25 if ((c & 0x80) == 0) return 1; // ASCII
26 if ((c & 0xE0) == 0xC0) return 2; // 110xxxxx
27 if ((c & 0xF0) == 0xE0) return 3; // 1110xxxx
28 if ((c & 0xF8) == 0xF0) return 4; // 11110xxx
29 return 0; // invalid byte
30}

32static inline int utf8_copy_truncate(size_t size, char *destination, const char *source) {
33 if (!destination || !source) return DESCENT_ERROR_NULL;
34 if (!size) return DESCENT_ERROR_OVERFLOW;
35
36 int result = 0;
37
38 size_t length;
39 size_t c_length;
40 for (length = 0; source[length]; length += c_length) {
41 c_length = utf8_char_length((unsigned char) source[length]);
42
43 if (!c_length) {
44 result = DESCENT_ERROR_INVALID;
45 break; // invalid UTF-8 byte
46 }
47 if (length + c_length >= size) {
48 result = DESCENT_WARN_TRUNCATION;
49 break; // prevent buffer overflow
50 }
51 }
52
53 memcpy(destination, source, length);
54 destination[length] = '\0';
55
56 return result;
57}
