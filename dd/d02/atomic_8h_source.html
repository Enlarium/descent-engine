127static inline int atomic_compare_exchange_32(atomic_32 *object, uint32_t *expected, uint32_t desired) {
128#if defined(DESCENT_PLATFORM_TYPE_POSIX)
129 return atomic_compare_exchange_strong(object, expected, desired);
130#elif defined(DESCENT_PLATFORM_TYPE_WINDOWS)
131 uint32_t original = (uint32_t) InterlockedCompareExchange((LONG*)object, (LONG)desired, (LONG)*expected);
132 if (original == *expected) return 1;
133 else {
134 *expected = original;
135 return 0;
136 }
137#endif
138}

147static inline int atomic_compare_exchange_64(atomic_64 *object, uint64_t *expected, uint64_t desired) {
148#if defined(DESCENT_PLATFORM_TYPE_POSIX)
149 return atomic_compare_exchange_strong(object, expected, desired);
150#elif DESCENT_WINDOWS_INTERLOCKED_64 == 1
151 uint64_t original = (uint64_t) InterlockedCompareExchange64((LONG64*)object, (LONG64)desired, (LONG64)*expected);
152 if (original == *expected) return 1;
153 else {
154 *expected = original;
155 return 0;
156 }
157#elif DESCENT_WINDOWS_INTERLOCKED_64 == 0
158 mutex_lock(&object->lock);
159 if (object->value == *expected) {
160 object->value = desired;
161 mutex_unlock(&object->lock);
162 return 1;
163 } else {
164 *expected = object->value;
165 mutex_unlock(&object->lock);
166 return 0;
167 }
168#endif
169}
