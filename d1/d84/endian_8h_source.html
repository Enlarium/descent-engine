27// Determines whether the platform is little-endian
28DESCENT_BUILTIN_API int
29is_little_endian() {
30#if defined(__BYTE_ORDER__) && defined(__ORDER_LITTLE_ENDIAN__)
31 return __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__;
32#elif defined(_WIN32) || defined(__LITTLE_ENDIAN__) || defined(__ARMEL__) || defined(_MIPSEL)
33 return 1;
34#elif defined(__BIG_ENDIAN__) || defined(__ARMEB__) || defined(_MIPSEB)
35 return 0;
36#else
37 DESCENT_BUILTIN_CONST union {
38 uint32_t i;
39 uint8_t c[4];
40 } TEST = { 0x01020304 };
41
42 return TEST.c[0] == 0x04;
43#endif
44}
